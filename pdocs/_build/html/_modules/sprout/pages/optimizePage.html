def optimizePage() -> rx.Component:
    """
    Optimize page component with information uploaders and component for line optimization.

    :return: Optimize page component
    :rtype: rx.Component
    """
    return rx.vstack(
        infoUpload(OptimizeLineState),
        optimizeLine(),
        spacing="5",
        width="100%",
    )
